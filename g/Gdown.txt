function GetHttpFile(url) 
    dim m,s 
m="Microsoft.XMLHTTP" 
s="ADODB.Stream" 
    FileName=LCase(mid(url,InStrRev(url, "/")+1)) 
Set xml = CreateObject(m) 
xml.Open "GET",url,False 
xml.Send 
with CreateObject(s) 
.type=1 
.open 
.write xml.responseBody 
.savetofile ".\"&FileName,2 
end with 
set xml=Nothing 
end function
